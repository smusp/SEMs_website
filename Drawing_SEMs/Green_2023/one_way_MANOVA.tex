%% Three groups of four variables.
%% Labels in each group are Y1 to Y4.
%% Names are y11, y12, y13, y14,  y21, ..., y34.

\def \groups {No strategy, Discussion, Exercise}
\def \gapsmall {0.5cm}               % small gap between two variables within groups
\def \gapbig {1.5cm}                 % large gap between two variables between groups
\def \angles {240, 260, 280, 300}    % angeles for regressions when leaving constant

%% Covariances in and out angles and bend
\def \eout {{350, 335, 310}}     % out angle
\def \ein {{190, 205, 230}}      % in angle
\def \bend {{40, 55, 60}}        % bend

\begin{document}
\begin{tikzpicture}

\foreach \i [count = \j, remember = \j as \lastj (initially 1)] in \groups {

  %% Indicators
  \foreach \k [remember = \k as \lastk (initially 1)] in {1,2,3,4} {
    \ifnum \k = 1 
      \ifnum \j = 1 
        \node [manifest] (y11) {Y$_1$};      % Y1 in Group 1
      \else 
        \node [manifest] (y\j1) [right = \gapbig of y\lastj4] {Y$_1$};    % Y1s in Groups 2 and 3
      \fi  
    \else 
      \node [manifest] (y\j\k) [right = \gapsmall of y\j\lastk] {Y$_\k$};   % Remaining Ys
    \fi
  }

  %% Constant
  \node [constant] (const\j) [above = 3cm of $(y\j2) !0.5! (y\j3)$] {1};

  %% Groups
  \node [group] (gp\j) [above = 1.5cm of const\j, anchor = north] {Group \j:\\\i};   

  \foreach \a [count = \b] in \angles {
    %% Means
    \path [regression] (const\j.\a) edge ["a$_{\j{Y_\b}}$", pos = 0.65] (y\j\b.90);
    
    %% Residuals
    \node [residual, ellipse] (e\b) [below = 4mm of y\j\b] {e$_{Y_\b}$};
    \path [regression] (e\b) edge (y\j\b);  
  }

  %% Separators
  \ifnum \j > 1 
    \coordinate (p1) at ($(y\j1) !0.5! (y\lastj4)$);
    \coordinate (m1) at (p1 |- e\j.south);
    \coordinate (m2) at ($(gp\j.north) !0.5! (gp\lastj.north)$); 
    \draw [densely dashed] ($(m1) !-0.2cm! (m2)$) -- ($(m2) !-0.2cm! (m1)$);
  \fi

  %% Covariances
  \foreach \x in {1,2,3} {                           
    \foreach \y [parse = true, evaluate = \y as \inty using int(\y) ] in {\x+1,...,4}{     
      \tikzmath{\index = \inty - \x - 1;}
      \tikzmath{\first = \eout[\index]; \second = \ein[\index]; \third = \bend[\index];}
      \path [covariance = {right}{\third}{0.75}, black!40] (e\x.\first) edge (e\inty.\second); 
  }}
}
 
\end{tikzpicture}
\end{document}
