%% Three groups of four indicators.
%% Labels in each group are Y1 to Y4.
%% Names are y11, y12, y13, y14,  y21, ..., y34.

\def \groups {No strategy, Discussion, Exercise}
\def \gapsmall {0.5cm}           % small gap between indicators
\def \gapbig {1.5cm}             % big gap between groups
\def \angles {270/240, 282/253, 294/263, 306/270}  % angles for intercepts and loadings
                                                   % when leaving constant and latent

\begin{document}
\begin{tikzpicture}

\foreach \i [count = \j, remember = \j as \lastj (initially 1)] in \groups {

  %% Indicators
  \foreach \k [remember = \k as \lastk (initially 1)] in {1,2,3,4} {
    \ifnum \k = 1 
      \ifnum \j = 1 
        \node [manifest] (y11) {Y$_1$};      % Y1 in Group 1
      \else 
        \node [manifest] (y\j1) [right = \gapbig of y\lastj4] {Y$_1$};    % Y1s in Groups 2 and 3
      \fi  
    \else 
      \node [manifest] (y\j\k) [right = \gapsmall of y\j\lastk] {Y$_\k$};   % Remaining Ys
    \fi
  }

  %% Latent and its variance (above last indicator in group)
  \node [latent] (latent) [above = 3cm of y\j4] {LS};
  \path [variance = {left}{110}{3}] (latent.120) edge ["d" {above = 2pt, name = D}] (latent.60);

  %% Mean (above first indicator in group)
  \node [constant] (const) at (y\j1 |- latent) {1};   
  \path[regression] (const) edge ["\ifnum \j = 1 {0} \else {a$_\j$} \fi"] (latent);

  %% Groups
  \coordinate (mp) at ($(y\j2) !0.5! (y\j3)$) ;
  \node [group] (gp\j) [above = 1cm of mp |- D, anchor = north] {Group \j:\\\i};   

  \foreach \a/\b [count = \c] in \angles {
    %% Residuals
    \node [residual={1pt}{0}{}] (e\c) [below = 4mm of y\j\c] {e$_{Y_\c}$};
    \path [regression] (e\c) edge (y\j\c);  

    %% Intercepts
    \path [regression] (const.\a) edge ["$\uptau_\c$", 
      pos = 0.5 - 1/(6*(5-\c))] (y\j\c.90);     %  A little less than .5, moreso for longer lines

    %% Loadings
    \path[regression] (latent.\b) edge ["\ifnum \c = 1 {1} \else {$\uplambda_\c$} \fi", 
      pos = 0.5 - 1/(6*\c)] (y\j\c.90);

  %% Separators
  \ifnum \j > 1 
    \coordinate (p1) at ($(y\j1) !0.5! (y\lastj4)$);
    \coordinate (m1) at (p1 |- e\j.south);
    \coordinate (m2) at ($(gp\j.north) !0.5! (gp\lastj.north)$); 
    \draw [densely dashed] ($(m1) !-0.2cm! (m2)$) -- ($(m2) !-0.2cm! (m1)$);
  \fi
}}

\end{tikzpicture}
\end{document}
