\newcommand {\drawfactor} [3] {    
                           %  #1 pos or neg indicator 
                           %  #2 POS or NEG factor 
                           %  #3 position 1st indicator
  %% Indicators
  \node [manifest] (#11) [#3] {$#1_1$};
  \node [manifest] (#12) [right = 1cm of #11] {$#1_2$};
  \node [manifest] (#13) [right = 1cm of #12] {$#1_3$};

  %% Factor
  \node [latent] (#2) [above = 2cm of #12] {#2};

  \foreach \i in {1,2,3} {
    %% Loadings  
    \path [regression] (#2) edge (#1\i.90);
    
    %% Residuals
    \node [residual] (e\i) [below right = 4mm of #1\i] {};
    \path [regression] (e\i) edge (#1\i.south east);
  }
}
