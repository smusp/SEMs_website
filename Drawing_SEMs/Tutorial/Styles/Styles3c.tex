\newcommand {\drawmodel} [1] {
  \begin{scope} [local bounding box = mybox]
  %% Manifest
  \node [manifest] (x) {X};
  \node [manifest] (y) [right = 2cm of x] {Y};

  %% Regression
  \path [regression] (x) edge (y);
  
  \node [below = 5pt, anchor = north west, inner sep = 0] at (mybox.south west) {#1};
  \end{scope}
}
