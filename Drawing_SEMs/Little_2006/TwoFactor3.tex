\documentclass[border = 10pt]{standalone}
\input{../SEMstyles.tex}
\usetikzlibrary{matrix}

\def \gapsmall {1cm}              % small gap between indicators
\def \gapbig {2cm}                % big gap between factors
\def \anglec {110, 90, 70}        % angles for intercepts leaving constant

\newcommand {\drawfactor} [3] {    %%   #1  latent: POS, NEG
                                   %%   #2  indicator: pos neg
                                   %%   #3  position of 1st indicator

  %% Indicators
  \node [manifest] (#21) [#3] {$#2_1$};
  \node [manifest] (#22) [right = \gapsmall of #21] {$#2_2$};
  \node [manifest] (#23) [right = \gapsmall of #22] {$#2_3$};

  %% What changes between factors:
  %%    latent variances - phi
  %%    counts for loadings, residuals, intercepts - count
  \ifnum \pdfstrcmp{#1}{POS}=0
    \def \phi {11}
    \def \count {{1, 2, 3}}
  \else
    \def \phi {22}
    \def \count {{4, 5, 6}}
  \fi

  %% Latent
  \node [latent] (#1) [above = 3cm of #22] {$#1$};
 
  %% Latent variance
  \path [variance = {left}{100}{2.5}] (#1.120) edge ["$\upphi_{\phi}$" {above = 1pt}] (#1.60);
  
  %% Constant for intercepts
  \node [constant, black!30] (M#2) [below = 2cm of #22] {1};  
  
  %%
  \foreach \i [count = \j] in \anglec {
    \tikzmath{\num = \count[\j-1];}
    
    %% Loadings
    \path [regression] (#1) edge ["$\uplambda_{\num}$"] (#2\j.90);
    
    %% Intercepts
    \path [regression, black!30] (M#2.\i) edge ["$\uptau_\num$"] (#2\j.270);   

    %% Residuals
    \node [residual] (e\j) [above right = 4mm of #2\j] {$\uptheta_\num$};
    \path [regression] (e\j) edge (#2\j.north east);
  }
}

\begin{document}
\begin{tikzpicture}

\drawfactor{POS}{pos}{}
\drawfactor{NEG}{neg}{right = \gapbig of pos3}

%% Latent means
\node [constant, black!30] (M) at ($(POS) !0.5! (NEG)$) {1};   
\path [regression, black!30] (M) edge ["$\upkappa_1$"] (POS) 
                             (M) edge ["$\upkappa_2$"] (NEG); 

%% Latent covariance
\path [covariance = {left}{35}{0.7}] (POS.30) edge ["$\upphi_{12}$"] (NEG.150);

%% Constraints
\matrix [matrix of nodes, draw, cells = {anchor = west}, anchor = east] at (POS -| pos1) {
   \bf{Constraints}\\
   $\uplambda_1 + \uplambda_2 + \uplambda_3 = 3$ \\
   $\uplambda_4 + \uplambda_5 + \uplambda_6 = 3$ \\ [2ex]

   $\uptau_1 + \uptau_2 + \uptau_3 = 0$ \\
   $\uptau_4 + \uptau_5 + \uptau_6 = 0$ \\
};

\end{tikzpicture}
\end{document}
