\def \prefac {{"POS", "NEG"}}         % prefix for factors
\def \preint {{"pos", "neg"}}         % prefix for indicators
\tikzmath{\fa = \prefac[0]; \fb = \prefac[1];}
\tikzmath{\ia = \preint[0]; \ib = \preint[1];}

\def \gapsmall {1.5cm}            % small gap between indicators
\def \gapbig {2.5cm}              % big gap between factors
\def \anglec {110, 90, 70}        % angles for intercepts leaving constant

%% Macro to draw a one-factor model
\newcommand{\FactorModel}[4]{     %%   #1  latent: POS, NEG
                                  %%   #2  indicator: pos, neg
                                  %%   #3  grade: 7, 8
                                  %%   #4  position of first indicator  
                                 
  %% Indicators
  \node [manifest] (#2#31) [#4] {$#2_1$};
  \node [manifest] (#2#32) [right = \gapsmall of #2#31] {$#2_2$};
  \node [manifest] (#2#33) [right = \gapsmall of #2#32] {$#2_3$};

  %% What changes beween factors:
  %%    latent variances - phi
  %%    counts for loadings, residuals, intercepts - count
  \ifnum\pdfstrcmp{#1}{\fa} = 0
    \def \phi {11}    \def \count {{1, 2, 3}}
  \else
    \def \phi {22}    \def \count {{4, 5, 6}}
  \fi
  
  %% Latent
  \node [latent] (#1#3) [above = 3cm of #2#32] {$#1$};

  %% Latent variance - differs according to grade
  \ifnum #3 = 7 \def \lab {$\upphi_{#3,\phi}\!=\!1$} \else \def \lab {$\upphi_{#3,\phi}$} \fi
  \path [variance = {left}{100}{2.5}] (#1#3.120) edge ["\lab" {above = 2pt}] (#1#3.60);

  %% Constant
  \node [constant, black!30] (M#2#3) [below = 1.5cm of #2#32] {1}; 

  \foreach \i [count = \j] in \anglec {
    \tikzmath{\num = \count[\j-1];}

    %% Loadings
    \path [regression] (#1#3) edge ["$\uplambda_{\num}$"] (#2#3\j.90);
    
    %% Intercepts
    \path [regression, black!30] (M#2#3.\i) edge ["$\uptau_\num$"] (#2#3\j.270);
    
    %% Residuals
    \node [residual] (e\j) [above right = 4mm of #2#3\j] {$\uptheta_{#3,\num}$};
    \path [regression] (e\j.south west) edge (#2#3\j.north east);
  }  
}

%% Macro to draw model for grade - two factors side-by-side
\newcommand{\GradeModel}[2]{            %%   #1  grade: 7, 8
                                        %%   #2  position eg {below = 9cm of pos71}

  %% Position factor models
  \FactorModel{\fa}{\ia}{#1}{#2}
  \FactorModel{\fb}{\ib}{#1}{right = \gapbig of \ia#13}
  
  %% Latent covariance
  \path [covariance = {left}{35}{0.8}] (\fa#1.30) edge ["$\upphi_{#1,12}$" {name = C1}] (\fb#1.150);
  
  %% Latent means - differ according to grade
  \ifnum #1 = 7  \def \laba {$\upkappa_{#1,1}\!=\!0$}   \def \labb {$\upkappa_{#1,2}\!=\!0$} 
    \else \def \laba {$\upkappa_{#1,1}$}   \def \labb {$\upkappa_{#1,2}$}   \fi
  \node [constant, black!30] (M) at ($(\fa#1) !0.5! (\fb#1)$) {1}; 
  \path [regression, black!30] (M) edge ["\laba"] (\fa#1) 
                               (M) edge ["\labb"] (\fb#1); 
}
