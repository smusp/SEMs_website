\documentclass[border = 10pt]{standalone}

\input{../SEMstyles.tex}
\usepackage{xstring}

\begin{document}
\begin{tikzpicture}

 
\def \labname {pos, neg}
\def \labnum {1, 2, 3}
\def \list {}
\foreach \x in \labname {
   \foreach \y in \labnum {
   \xdef\list{\list \x$_\y$,}
}}
\StrGobbleRight{\list}{1}[\list]

%% pos and neg manifest
\foreach \i [count = \j, remember = \i as \lasti] in \list {
   \ifnum \j = 1
     \node [manifest] (\i) at (\j, 0) {\i};
     \node [residual2] (e\j) [above left = 4mm of \i] {$\uptheta_\j$};
     \path [regression] (e\j) edge (\i.north west);
   \else
     \ifnum \j = 4 \def \gap {3} \else \def \gap {1.6} \fi
     \ifnum \j < 4 \def \angle {north west} \else \def \angle {north east} \fi
     \ifnum \j < 4 \def \pos {left} \else \def \pos {right} \fi     
     \node[manifest] (\i) [right = \gap cm of \lasti] {\i};
     \node [residual2] (e\j) [above \pos = 4mm of \i] {$\uptheta_\j$};
     \path [regression] (e\j) edge (\i.\angle);
   \fi
}

 
\end{tikzpicture}
