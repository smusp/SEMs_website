\usetikzlibrary{matrix}

\def \anglec {110, 90, 70}        % angles for intercepts leaving constant
\def \gapsmall {1.5cm}            % small gap between two variables within groups
\def \gapbig {3cm}                % large gap between two variables between groups

\newcommand {\drawgrade} [2] {    % #1  grade: 7, 8;  #2  position 1st variable
  %% manifest
  \node [manifest] (pos#11) [#2] {$pos_1$};
  \node [manifest] (pos#12) [right = \gapsmall of pos#11] {$pos_2$};
  \node [manifest] (pos#13) [right = \gapsmall of pos#12] {$pos_3$};

  %% Latent and its variance
  \node [latent] (POS#1) [above = 3cm of pos#12] {$POS$};
  \path [variance = {left}{100}{2.5}] (POS#1.120) edge ["$\upphi_#1$" {above = 2pt}] (POS#1.60);

  %% Constants and latent mean
  \node [constant, black!30] (M#11) [right = 2cm of POS#1] {1};
  \node [constant, black!30] (M#12) [below = 2cm of pos#12] {1};
  \path [regression, black!30] (M#11) edge ["$\upkappa_#1$"] (POS#1);

  %% Grades
  \node [group] (gp#1) [above = 1.5cm of POS#1] {Grade #1};

  \foreach \i [count = \j] in \anglec {
  %% Loadings
  \path [regression] (POS#1) edge ["$\uplambda_\j$"] (pos#1\j.90);

  %% Residuals
  \node [residual] (e#1\j) [above left = 4mm of pos#1\j] {$\uptheta_{#1,\j}$};
  \path [regression] (e#1\j.south east) edge (pos#1\j.north west);

  %% Intercepts
  \path [regression, black!30] (M#12.\i) edge ["$\uptau_\j$"] (pos#1\j.270);
  }
}
