\input{../SEMstyles.tex}

\begin{document}
\begin{tikzpicture}

%% pos manifest
\node [manifest] (pos1) {$pos_1$};
\node [manifest] (pos2) [right = 1.5cm of pos1] {$pos_2$};
\node [manifest] (pos3) [right = 1.5cm of pos2] {$pos_3$};

%% neg manifest
\node [manifest] (neg1) [right = 3cm of pos3] {$neg_1$};
\node [manifest] (neg2) [right = 1.5cm of neg1] {$neg_2$};
\node [manifest] (neg3) [right = 1.5cm of neg2] {$neg_3$};

%% POS and NEG latents
\node [latent] (POS) [above = 3cm of pos3] {$POS$};
\node [latent] (NEG) [above = 3cm of neg1] {$NEG$};

%% Latent variances and covariance
\path [covariance] (POS.35) edge ["$\upphi_{12}$", bend left = 40] (NEG.145);
\path [variance] (POS.120) edge ["$\upphi_{11}\!=\!1$", above, outer sep = 1pt,
   bend left = 100, looseness = 2.5] (POS.60);
\path [variance] (NEG.120) edge ["$\upphi_{22}\!=\!1$", above, outer sep = 1pt,
   bend left = 100, looseness = 2.5] (NEG.60);

%% Latent means
\node [constant, black!30] (M) [below = 1.5cm of $(pos3) !0.5! (neg1)$] {1};
\path [regression, black!30] (M) edge ["$\upkappa_1\!=\!0$", pos = 0.6] (POS);
\path [regression, black!30] (M) edge ["$\upkappa_2\!=\!0$", pos = 0.6] (NEG);

%% Loadings, Residuals, and Intercepts for POS Factor 
\def \angleMean {170, 150, 130}   % angles for intercept out of constant
\foreach \i [count = \j] in \angleMean {
%% Loadings
\path [regression] (POS) edge ["$\uplambda_\j$"] (pos\j.90);

%% Intercepts
\path [regression, black!30] (M.\i) edge ["$\uptau_\j$"] (pos\j.south east);

%% Residuals
\node [residual2] (e\j) [above left = 4mm of pos\j] {$\uptheta_\j$};
\path [regression] (e\j) edge (pos\j.north west);
}

%% Loadings, Residuals, and Intercepts for NEG factor
\def \angleMean {50, 30, 10}   % angles for intercept out of constant
\foreach \i [count = \j, evaluate = \j as \k using int(\j + 3)] in \angleMean {
%% Loadings
\path [regression] (NEG) edge ["$\uplambda_\k$"] (neg\j.90);

%% Intercepts
\path [regression, black!30] (M.\i) edge ["$\uptau_\k$"] (neg\j.south west);

%% Residuals
\node [residual2] (e\k) [above right = 4mm of neg\j] {$\uptheta_\k$};
\path [regression] (e\k) edge (neg\j.north east);
}

\end{tikzpicture}
\end{document}
